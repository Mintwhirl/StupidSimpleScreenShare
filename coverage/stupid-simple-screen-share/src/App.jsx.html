                  </div>
                )}

                {showDiagnostics && currentView !== 'home' && (
                  <div className='fixed left-0 top-16 bottom-0 w-80 bg-black bg-opacity-80 backdrop-blur-sm shadow-lg border-r border-purple-500 border-opacity-30 z-50'>

                    </Suspense>
                  </div>
                )}
          </div>
        )}
      </div>
    </div>
  );
}
